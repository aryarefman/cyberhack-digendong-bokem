// WeeklyStockChart component using recharts LineChart
function WeeklyStockChart({ data }) {
  if (!data || data.length === 0) {
    return (
      <div className="dashboard-chart-empty">
        <p>No trend data available</p>
      </div>
    );
  }

  // Format date labels for X-axis (e.g., "Jan 20")
  const formatDate = (dateStr) => {
    const date = new Date(dateStr + 'T00:00:00');
    return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
  };

  const chartData = data.map(item => ({
    ...item,
    label: formatDate(item.date)
  }));

  return (
    <div className="dashboard-chart-card">
      <div className="dashboard-chart-card-header">
        <h3 className="dashboard-chart-card-title">Weekly Stock Trend</h3>
      </div>
      <div className="dashboard-chart-wrapper">
        <ResponsiveContainer width="100%" height={280}>
          <LineChart data={chartData} margin={{ top: 10, right: 30, left: 0, bottom: 0 }}>
            <CartesianGrid strokeDasharray="3 3" stroke="rgba(0,0,0,0.06)" />
            <XAxis
              dataKey="label"
              tick={{ fontSize: 12, fill: '#737969' }}
              axisLine={{ stroke: '#C2C9B6' }}
              tickLine={false}
            />
            <YAxis
              tick={{ fontSize: 12, fill: '#737969' }}
              axisLine={{ stroke: '#C2C9B6' }}
              tickLine={false}
              allowDecimals={false}
            />
            <Tooltip
              contentStyle={{
                background: '#fff',
                border: '1px solid rgba(143,177,87,0.3)',
                borderRadius: '8px',
                fontSize: '13px'
              }}
              labelFormatter={(label) => `Date: ${label}`}
              formatter={(value) => [`${value} items`, 'Active Stock']}
            />
            <Line
              type="monotone"
              dataKey="count"
              stroke="#366306"
              strokeWidth={2.5}
              dot={{ fill: '#366306', strokeWidth: 2, r: 4 }}
              activeDot={{ r: 6, fill: '#366306' }}
            />
          </LineChart>
        </ResponsiveContainer>
      </div>
    </div>
  );
}

// ZoneSummaryCards component showing each zone's item count and capacity percent
function ZoneSummaryCards({ data }) {
  if (!data || data.length === 0) {
    return (
      <div className="dashboard-zone-empty">
        <p>No zone data available</p>
      </div>
    );
  }

  return (
    <div className="dashboard-zone-cards">
      <div className="dashboard-zone-cards-header">
        <h3 className="dashboard-zone-cards-title">Zone Summary</h3>
      </div>
      <div className="dashboard-zone-grid">
        {data.map((zone) => (
          <div key={zone.zone} className="dashboard-zone-card">
            <div className="zone-card-header">
              <span className="zone-card-label">Zone {zone.zone}</span>
              <span className="zone-card-count">{zone.itemCount} items</span>
            </div>
            <div className="zone-card-progress-track">
              <div
                className={`zone-card-progress-fill ${zone.capacityPercent >= 90 ? 'zone-card-progress-fill--danger' : zone.capacityPercent >= 70 ? 'zone-card-progress-fill--warning' : ''}`}
                style={{ width: `${zone.capacityPercent}%` }}
              ></div>
            </div>
            <span className="zone-card-capacity">{zone.capacityPercent}% capacity</span>
          </div>
        ))}
      </div>
    </div>
  );
}

// ExpiryAlertList component showing top 5 nearest-expiry items
function ExpiryAlertList({ data }) {
  if (!data || data.length === 0) {
    return (
      <div className="dashboard-expiry-card">
        <div className="dashboard-expiry-card-header">
          <h3 className="dashboard-expiry-card-title">Expiry Alerts</h3>
        </div>
        <div className="dashboard-expiry-empty">
          <p>No expiry alerts</p>
        </div>
      </div>
    );
  }

  return (
    <div className="dashboard-expiry-card">
      <div className="dashboard-expiry-card-header">
        <h3 className="dashboard-expiry-card-title">Expiry Alerts</h3>
        <span className="dashboard-expiry-count">{data.length} items</span>
      </div>
      <div className="dashboard-expiry-list">
        {data.map((item) => (
          <div key={item.id} className="dashboard-expiry-item">
            <div className="expiry-item-info">
              <span className="expiry-item-name">{item.name}</span>
              <span className="expiry-item-zone">Zone {item.zone}</span>
            </div>
            <span className={`expiry-item-days ${item.daysLeft <= 7 ? 'expiry-item-days--critical' : item.daysLeft <= 14 ? 'expiry-item-days--warning' : ''}`}>
              {item.daysLeft} days left
            </span>
          </div>
        ))}
      </div>
    </div>
  );
}

export default function DashboardPage() {
  const { user } = useAuth();
  
  // Dynamic database states
  const [inventory, setInventory] = useState([]);
  const [activities, setActivities] = useState([]);
  const [coldChainAlertsCount, setColdChainAlertsCount] = useState(0);
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState(null);

  // Dashboard stats from /api/dashboard/stats
  const [dashboardStats, setDashboardStats] = useState(null);
  const [statsLoading, setStatsLoading] = useState(true);

  // Fetch Dashboard Data from live PostgreSQL database
  const fetchDashboardData = async () => {
    try {
      setIsLoading(true);
      setError(null);
      const [invRes, auditRes, tempRes, statsRes] = await Promise.all([
        fetch('/api/inventory'),
        fetch('/api/audit'),
        fetch('/api/cold-chain'),
        fetch('/api/dashboard/stats')
      ]);

      const invData = await invRes.json();
      const auditData = await auditRes.json();
      const tempData = await tempRes.json();
      const statsData = await statsRes.json();

      if (invData.success) {
        setInventory(invData.items);
      }
      if (auditData.success) {
        setActivities(auditData.logs.slice(0, 5));
      }
      if (tempData.success) {
        const zoneDReadings = tempData.temperatures.D || [];
        const zoneDAnomalies = zoneDReadings.filter(r => r.temperature > 5 || r.temperature < -5);
        setColdChainAlertsCount(zoneDAnomalies.length > 0 ? 1 : 0);
      }
      if (statsData.success) {
        setDashboardStats(statsData);
      }
    } catch (err) {
      console.error('Error loading dashboard data:', err);
      setError('Unable to connect to the server. Please try again later.');
    } finally {
      setIsLoading(false);
      setStatsLoading(false);
    }
  };

  useEffect(() => {
    Promise.resolve().then(fetchDashboardData);
  }, []);

  // Live Computed Stats
  const stats = useMemo(() => {
    const totalSlots = 30;
    const occupiedSlots = inventory.length;
    const capacity = Math.round((occupiedSlots / totalSlots) * 100);

    const expired = inventory.filter(i => i.status === 'Expired').length;
    const critical = inventory.filter(i => i.status === 'Kritis').length;
    const warning = inventory.filter(i => i.status === 'Warning').length;
    const safe = inventory.filter(i => i.status === 'Aman').length;

    return { capacity, totalSlots, occupiedSlots, expired, critical, warning, safe };
  }, [inventory]);

  // Loading State
  if (isLoading) {
    return (
      <div className="dashboard animate-fade dashboard-loading-wrapper">
        <div className="dashboard-loading-content">
          <div className="dashboard-loading-spinner"></div>
          <div className="dashboard-loading-text-wrapper">
            <p className="dashboard-loading-title">Loading Dashboard</p>
            <p className="dashboard-loading-subtitle">Fetching live warehouse data...</p>
          </div>
        </div>
      </div>
    );
  }

  // Error State
  if (error) {
    return (
      <div className="dashboard animate-fade dashboard-loading-wrapper">
        <div className="dashboard-loading-content">
          <AlertTriangle size={48} style={{ color: 'var(--color-status-critical, #BA1A1A)' }} />
          <div className="dashboard-loading-text-wrapper">
            <p className="dashboard-loading-title">Failed to Load Dashboard</p>
            <p className="dashboard-loading-subtitle">{error}</p>
          </div>
          <button className="btn btn-primary" onClick={fetchDashboardData} style={{ marginTop: '16px' }}>
            Retry
          </button>
        </div>
      </div>
    );
  }

  // Count active items
  const activeItemsCount = inventory.filter(i => i.status !== 'Expired').length;
  const expiringSoonCount = inventory.filter(i => i.status === 'Warning' || i.status === 'Kritis').length;

  // Items requiring immediate use (Expired or Kritis)
  const immediateUseItems = inventory
    .filter(i => i.status === 'Expired' || i.status === 'Kritis')
    .sort((a, b) => new Date(a.expiry) - new Date(b.expiry));

  // Fallback table data matching Figma exactly
  const fallbackTableData = [
    { id: 'LOT-PO-230526', name: 'Patchouli Oil', daysLeft: 5, daysLabel: '5 days left', status: 'Critical', statusClass: 'critical' },
    { id: 'LOT-VE-230610', name: 'Vetiver Extract', daysLeft: 14, daysLabel: '14 days left', status: 'Warning', statusClass: 'warning' },
    { id: 'LOT-BE-230701', name: 'Bergamot Essential', daysLeft: 28, daysLabel: '28 days left', status: 'Monitor', statusClass: 'monitor' }
  ];

  // Fallback activity data
  const fallbackActivities = [
    { id: 'act-1', timestamp: '09:42', user: 'Operator Budi', detail: 'Added 200kg Patchouli Oil to Slot B-04.' },
    { id: 'act-2', timestamp: '08:15', user: 'QC Staff Sari', detail: 'Approved Vanillin batch.' },
    { id: 'act-3', timestamp: '08:00', user: 'Operator Budi', detail: 'Added 100kg Patchouli Oil to Slot B-03.' },
    { id: 'act-4', timestamp: '07:45', user: 'QC Staff Sari', detail: 'Approved Vanillin batch.' }
  ];

  // Process activities for display
  const displayActivities = activities.length > 0
    ? activities.map((log) => {
        let timeStr = '09:42';
        if (log.timestamp) {
          const parts = log.timestamp.split(' ');
          timeStr = parts[1] || parts[0] || log.timestamp;
          if (timeStr.includes(':')) {
            timeStr = timeStr.slice(0, 5);
          }
        }
        return { id: log.id, timestamp: timeStr, user: log.user || log.username, detail: log.detail };
      })
    : fallbackActivities;

  return (

      <h1 className="dashboard-page-title">Dashboard</h1>

      {/* 4 Stat Cards Grid */}
      <div className="dashboard-stats-grid">

        {/* Card 1: Total Active Stock */}
        <div className="dashboard-stat-card">
          <div className="stat-card-top">
            <div className="stat-card-content">
              <span className="stat-card-label">Total active stock</span>
              <div className="stat-card-value-row">
                <span className="stat-card-value">{activeItemsCount || 248}</span>
                <span className="stat-card-unit">lots</span>
              </div>
            </div>
            <div className="stat-card-icon-box stat-card-icon-box--green">
              <Package size={24} className="stat-card-icon--dark" />
            </div>
          </div>
          <div className="stat-card-footer">
            <TrendingUp size={20} className="stat-card-trend-icon--positive" />
            <span className="stat-card-trend--positive">+12 this week</span>
          </div>
        </div>

        {/* Card 2: Nearing Expiry */}
        <div className="dashboard-stat-card">
          <div className="stat-card-top">
            <div className="stat-card-content">
              <span className="stat-card-label">Nearing Expiry</span>
              <div className="stat-card-value-row">
                <span className="stat-card-value">{expiringSoonCount || 24}</span>
                <span className="stat-card-unit">items</span>
              </div>
            </div>
            <div className="stat-card-icon-box stat-card-icon-box--red-light">
              <CalendarClock size={24} className="stat-card-icon--red" />
            </div>
          </div>
          <div className="stat-card-footer">
            <AlertTriangle size={14} className="stat-card-trend-icon--negative" />
            <span className="stat-card-trend--negative">within 30 days</span>
          </div>
        </div>

        {/* Card 3: Warehouse Capacity */}
        <div className="dashboard-stat-card">
          <div className="stat-card-top">
            <div className="stat-card-content">
              <span className="stat-card-label">Warehouse Capacity</span>
            </div>
            <div className="stat-card-icon-box stat-card-icon-box--green">
              <Warehouse size={24} className="stat-card-icon--dark" />
            </div>
          </div>
          <div className="stat-card-progress-wrapper">
            <div className="stat-card-progress-track">
              <div className="stat-card-progress-fill" style={{ width: `${stats.capacity}%` }}></div>
            </div>
            <p className="stat-card-progress-text">
              {stats.totalSlots - stats.occupiedSlots} empty slots remaining
            </p>
          </div>
        </div>

        {/* Card 4: Cold-Chain Alerts */}
        <div className="dashboard-stat-card">
          <div className="stat-card-top">
            <div className="stat-card-content">
              <span className="stat-card-label">Cold-Chain Alerts</span>
              <div className="stat-card-value-row">
                <span className="stat-card-value stat-card-value--danger">{coldChainAlertsCount || 2}</span>
                <span className="stat-card-unit">zones</span>
              </div>
            </div>
            <div className="stat-card-icon-box stat-card-icon-box--red">
              <Thermometer size={24} className="stat-card-icon--white" />
            </div>
          </div>
          <div className="stat-card-footer">
            <span className="stat-card-alert-text">
              temperature anomaly detected in storage C
            </span>
          </div>
        </div>

      </div>

      {/* Weekly Stock Trend Chart */}
      {statsLoading ? (
        <div className="dashboard-chart-card">
          <div className="dashboard-chart-card-header">
            <h3 className="dashboard-chart-card-title">Weekly Stock Trend</h3>
          </div>
          <div className="dashboard-chart-loading">
            <div className="dashboard-loading-spinner" style={{ width: 32, height: 32 }}></div>
            <p style={{ margin: 0, fontSize: 13, color: '#737969' }}>Loading chart data...</p>
          </div>
        </div>
      ) : (
        <WeeklyStockChart data={dashboardStats?.weeklyTrend} />
      )}

      {/* Quick Stats Section */}
      {statsLoading ? (
        <div className="dashboard-quick-stats">
          <div className="dashboard-quick-stats-header">
            <h3 className="dashboard-quick-stats-title">Quick Stats</h3>
          </div>
          <div className="dashboard-section-loading">
            <div className="dashboard-loading-spinner" style={{ width: 24, height: 24 }}></div>
            <p style={{ margin: 0, fontSize: 13, color: '#737969' }}>Loading...</p>

      )}

      {statsLoading ? (

          <div className="dashboard-zone-cards-header">

                  <>
                    {fallbackTableData.map(item => (
                      <tr key={item.id}>
                        <td>
                          <div className="table-material-cell">
                            <div className={`table-material-icon table-material-icon--${item.statusClass}`}>
                              <Droplets size={16} />
                            </div>

              </tbody>
            </table>
          </div>
        </div>

        {/* Right: Recent Activity */}
        <div className="dashboard-activity-card">
          <div className="dashboard-activity-card-header">
            <h3 className="dashboard-activity-card-title">Recent Activity</h3>
          </div>

          <div className="dashboard-activity-feed">
            {displayActivities.map((log, idx, arr) => (
              <div key={log.id} className="dashboard-activity-item">
                {/* Time Column */}
                <span className="activity-time-col">{log.timestamp}</span>

}
